## Quickstart

### Generate Text from Text Input
Send an input text to generate a response.

<CodeGroup>
```javascript javascript 
import Bytez from "bytez.js";

const client = new Bytez("YOUR_BYTEZ_KEY_HERE");
const model = client.model("google/flan-t5-base");

const modelParams = { max_new_tokens: 20, temperature: 2 };

const { error, output } = await model.run(
  "Once upon a time there was a small little man who",
  modelParams
);

if (error) {
  console.error("Error:", error);
} else {
  console.log("Generated text:", output[0].generated_text);
}
```

```python python
from bytez import Bytez

client = Bytez("YOUR_BYTEZ_KEY_HERE")

model = client.model("google/flan-t5-base")

model.load()

input_text = "Once upon a time there was a small little man who"

model_params = {"max_new_tokens": 20, "temperature": 2}

result = model.run(input_text, model_params=model_params)

output = result["output"]

generated_text = output[0]["generated_text"]

print(generated_text)
```

```julia julia
using Bytez

client = Bytez.init("YOUR_BYTEZ_KEY_HERE")

model = client.model("google/flan-t5-base")

model.load()

input_text = "Once upon a time there was a small little man who"

model_params = Dict("max_new_tokens" => 20, "temperature" => 2)

result = model.run(input_text, model_params)

output = result["output"]

generated_text = output[1]["generated_text"]

println(generated_text)
```
