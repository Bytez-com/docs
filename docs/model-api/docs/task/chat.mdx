	<Accordion defaultOpen="false" title="Stream text">
		
		<CodeGroup>
			```javascript javascript
			import Bytez from 'bytez.js';
			
			// insert your key
			const sdk = new Bytez('BYTEZ_KEY');
			
			// choose your model
			const model = sdk.model('Qwen/Qwen3-4B');
			
			// provide the model with input
			const input = [
			  {
			    "role": "system",
			    "content": "You are a friendly chatbot"
			  },
			  {
			    "role": "assistant",
			    "content": "Hello, I'm a friendly bot"
			  },
			  {
			    "role": "user",
			    "content": "Hello bot, what is the capital of England?"
			  }
			];
			
			// set streaming to "true"
			const stream = true;
			
			// send to the model
			const readStream = await model.run(input, stream);
			
			let text = '';
			
			for await (const chunk of readStream) {
			  text += chunk;
			  console.log(chunk);
			}
			
			// observe the output
			console.log({ text });
			
			```
			```python python
			from bytez import Bytez
			
			# insert your key
			sdk = Bytez("BYTEZ_KEY")
			
			# choose your model
			model = sdk.model("Qwen/Qwen3-4B")
			
			# provide the model with input
			input = [
			  {
			    "role": "system",
			    "content": "You are a friendly chatbot"
			  },
			  {
			    "role": "assistant",
			    "content": "Hello, I'm a friendly bot"
			  },
			  {
			    "role": "user",
			    "content": "Hello bot, what is the capital of England?"
			  }
			]
			
			# set streaming to "true"
			stream = True
			
			# send to the model
			readStream = model.run(input, stream)
			
			text = ""
			
			for chunk in readStream:
			    tokens = chunk.decode("utf-8")
			    text += tokens
			    print(tokens)
			
			# observe the output
			print({"text": text})
			
			```
